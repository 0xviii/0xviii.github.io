$dir = __DIR__ . '/papers';

function list($dirBase, $dirRel = '')
{
    $result = [];
    $path = $dirBase . ($dirRel ? DIRECTORY_SEPARATOR.$dirRel : '');

    $itens = scandir($path);
    foreach ($itens as $item) {
        if ($item === '.' || $item === '..') {
            continue;
        }

        $caminhoItem = $path . DIRECTORY_SEPARATOR . $item;
        $caminhoRelativo = ltrim($dirRel . '/' . $item, '/');

        if (is_dir($caminhoItem)) {
            $result = array_merge($result, list($dirBase, $dirRelativo));
        } else {
            $result[] = $caminhoRelativo;
        }
    }

    return $result;
}
